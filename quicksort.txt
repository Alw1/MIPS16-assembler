# Quicksort Program 

# r1 == Array Size 
addi $r1, $r1, 5

# r2 == Array Base Address
addi $r2, $r2, 20

# r3 == loop counter

# Other registers are used for temporary values

# Step 1: Load array values into memory
addi $r3, $r0, 32
addi $r4, $r0, 40
addi $r5, $r0, 1
addi $r6, $r0, 63
addi $r7, $r0, -21
sw $r3, 0($r2)
sw $r4, 1($r2)
sw $r5, 2($r2)
sw $r6, 3($r2)
sw $r7, 4($r2)


quickSort:  beq $r3, $r1, end
			
			
			
			
			addi $r3, $r3, 1


end: j end
